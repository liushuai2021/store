多表查询练习
------------------------------------------------------------------------------
/*
1. 查出至少有一个员工的部门。显示部门编号、部门名称、部门位置、部门人数。
*/

SELECT t.deptno,d.dname,d.loc,COUNT(*) FROM t_employees t JOIN t_dept d ON t.deptno=d.deptno GROUP BY t.deptno

/*
3. 列出所有员工的姓名及其直接上级的姓名。
*/

SELECT t.ename,d.ename FROM t_employees t,t_employees d WHERE t.MGR=d.empno

/*
4. 列出受雇日期早于直接上级的所有员工的编号、姓名、部门名称。
*/

SELECT t.deptno,t.ename,d.dname FROM(SELECT t.deptno,t.ename FROM t_employees t,t_employees d WHERE t.MGR=d.empno AND t.hiredate<d.hiredate) t
JOIN t_dept d
ON t.deptno=d.deptno

/*
5. 列出部门名称和这些部门的员工信息，同时列出那些没有员工的部门
*/
SELECT t.*,d.dname,d.loc FROM t_employees t RIGHT OUTER JOIN t_dept d ON t.deptno=d.deptno

/*
7. 列出最低薪金大于15000的各种工作及从事此工作的员工人数。
*/
SELECT job ,COUNT(*)FROM t_employees GROUP BY job HAVING MIN(sal)>15000

/*
8. 列出在公关部工作的员工的姓名，假定不知道公关部的部门编号。
*/
SELECT t.ename ,d.dname FROM t_employees t JOIN t_dept d ON t.deptno=d.deptno AND d.dname="公关部"

/*
9. 列出薪金高于公司平均薪金的所有员工信息，所在部门名称，上级领导，工资等级。
*/
SELECT t.*,d.dname FROM t_dept d
JOIN  (SELECT t.*,o.ename AS "上级" FROM t_employees t, t_employees o WHERE  t.MGR=o.empno ) t
ON t.deptno=d.deptno AND sal>(SELECT AVG(sal) FROM t_employees)

/*
10.列出与张飞从事相同工作的所有员工及部门名称。
*/
SELECT t.ename,d.dname FROM t_employees t JOIN  t_dept d ON t.deptno=d.deptno AND job=(SELECT job FROM t_employees WHERE ename="张飞")

/*
11.列出薪金高于在部门30工作的所有员工的薪金的员工姓名和薪金、部门名称。
*/

SELECT t.ename,d.dname,t.sal FROM t_employees t JOIN t_dept d ON t.deptno=d.deptno WHERE sal>(SELECT MAX(sal)FROM t_employees WHERE deptno=30)
